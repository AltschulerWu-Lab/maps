maps.fitters

Fitter functions are used to run classification pipelines used for MAP-scoring analysis. Fitters define data splits, how model(s) should be trained relative to these sample splits, how predictions should be generated relative to these splits.

All fitter functions operate on a Screen and BaseModel object. Some fitter functions additionally specify holdout cell lines—i.e., those that should be removed during training and only evaluated as model predictions.

Fitters return dictionaries containing the following keys: fitted: list of fitted models, predicted: DataFrame of predictions, importance: DataFrame of feature importances

*_mut fitters are wrappers that run an analysis for each mutational background. As a result, they return one dict per mutational background.

 35def leave_one_out_mut(screen: 'ScreenBase', model: BaseModel) -> Dict:
 36    """Wrapper for running leave-one-out fitter by mutational background.
 37    
 38    Binary classifiers for <mutation> vs WT will be run for all ALS mutational 
 39    backgrounds, excluding sporadics. Each mutation is trained separately using
 40    leave-one-out cross-validation on cell lines.
 41    
 42    Args:
 43        screen (ScreenBase): Screen object containing data and metadata.
 44        model (BaseModel): Model instance (SKLearnModel or PyTorchModel).
 45        
 46    Returns:
 47        Dict[str, Dict]: Dictionary keyed by mutation name, each containing:
 48            - `fitted` (list): List of fitted models
 49            - `predicted` (pl.DataFrame): Predictions with metadata
 50            - `importance` (pd.DataFrame or None): Feature importances
 51            - Additional keys depending on model type
 52    """
 53    assert screen.data is not None, "screen data not loaded"
 54    assert screen.metadata is not None, "screen metadata not loaded"
 55    
 56    metadata = merge_metadata(screen)
 57    mutations = metadata["Mutations"].unique()
 58    mutations = set(mutations) - set(["WT", "sporadic"])
 59    out = {}
 60    
 61    for m in mutations:
 62        print(f"Training {m}...")
 63        mutations_holdout = list(mutations - {m}) + ["sporadic"]
 64        holdout = get_mutation_celllines(metadata, mutations_holdout)
 65        out[m] = leave_one_out(screen, model, holdout)
 66         
 67    return out

 69def leave_one_out(screen: 'ScreenBase', model: BaseModel, holdout: List = []) -> Dict:
 70    """Wrapper for leave-one-out cross-validation.
 71    
 72    Dispatches to sklearn or pytorch implementation based on model type. Leaves
 73    one cell line out at a time, trains on remaining cell lines, and predicts
 74    on the held-out cell line plus any additional holdout cell lines specified.
 75    
 76    Args:
 77        screen (ScreenBase): Screen object containing data and metadata.
 78        model (BaseModel): Model instance (SKLearnModel or PyTorchModel).
 79        holdout (List[str], optional): Additional cell lines to hold out from 
 80            training. Defaults to [].
 81            
 82    Returns:
 83        Dict: Dictionary containing:
 84            - `fitted` (list): List of fitted models (one per cell line)
 85            - `predicted` (pl.DataFrame): Predictions with metadata
 86            - `importance` (pd.DataFrame or None): Feature importances
 87            - Additional keys for PyTorch models (scalers, training_lines)
 88    """
 89    metadata = merge_metadata(screen)
 90    holdout += get_mutation_celllines(metadata, ["sporadic"])
 91    
 92    if isinstance(model, SKLearnModel):
 93        return leave_one_out_sklearn(screen, model, holdout)
 94    elif isinstance(model, PyTorchModel):
 95        return leave_one_out_pytorch(screen, model, holdout)
 96    else:
 97        raise ValueError(f"Unsupported model type: {type(model)}")

100def leave_one_out_sklearn(
101    screen: "ScreenBase", 
102    model: BaseModel, 
103    holdout: List = []) -> Dict:
104    """Leave-one-out cross-validation for scikit-learn models.
105    
106    Iterates through each cell line, training on all other cell lines and 
107    predicting on the held-out cell line. Models are trained on well-averaged
108    features.
109    
110    Args:
111        screen (ScreenBase): Screen object with data and metadata.
112        model (BaseModel): SKLearnModel instance.
113        holdout (List[str], optional): Cell lines to exclude from training.
114            Defaults to [].
115            
116    Returns:
117        Dict: Dictionary containing:
118            - `fitted` (list): List of fitted model copies
119            - `predicted` (pl.DataFrame): Predictions joined with metadata
120            - `importance` (pd.DataFrame): Feature importances from each fold
121    """
122    assert screen.data is not None, "screen data not loaded"
123    assert screen.metadata is not None, "screen metadata not loaded"
124    y = screen.get_response() 
125    x = screen.get_data()
126    
127    # Get unique cell lines and exclude test cell lines
128    cell_lines = (
129        screen.metadata
130        .select("CellLines")
131        .unique()
132        .to_series()
133        .to_list()
134    )
135    
136    cell_lines = set(cell_lines) - set(holdout)
137    
138    fitted, predicted, importance = [], [], []
139     
140    for cl in cell_lines: 
141        cell_lines_test = [cl]
142        cell_lines_test = cell_lines_test + holdout
143        
144        # Create train and test indices
145        id_train = screen.metadata \
146            .filter(~pl.col("CellLines").is_in(cell_lines_test)) \
147            .select("ID") \
148            .to_series()
149
150        id_test = screen.metadata \
151            .filter(pl.col("CellLines").is_in(cell_lines_test)) \
152            .select("ID") \
153            .to_series()
154        
155        # Fit model and make predictions
156        model.fit(x=x, y=y, id_train=id_train)
157        predicted_cl = model.predict(x=x, id_test=id_test)
158        predicted_cl= predicted_cl.with_columns(pl.lit(cl).alias("Holdout"))
159        
160        # Merge results
161        predicted.append(predicted_cl.join(screen.metadata, on="ID"))
162        fitted.append(copy.deepcopy(model))
163        importance.append(model.get_importance(x))
164      
165    out = {}  
166    out["fitted"] = fitted
167    out["predicted"] = pl.concat(predicted)
168    out["importance"] = pd.concat([pd.Series(p) for p in importance], axis=1)
169    return out
170

172def leave_one_out_pytorch(
173    screen: "ScreenBase", 
174    model: BaseModel, 
175    holdout: List = []) -> Dict:
176    """Leave-one-cell-line-out classification for PyTorch models.
177    
178    Trains PyTorch models using DataLoader batching on single-cell data.
179    Each iteration holds out one cell line for testing while training on all
180    remaining cell lines.
181    
182    Args:
183        screen (ScreenBase): Screen object (ImageScreen or ImageScreenMultiAntibody).
184        model (BaseModel): PyTorchModel instance.
185        holdout (List[str], optional): Cell lines to exclude from training.
186            Defaults to [].
187            
188    Returns:
189        Dict: Dictionary containing:
190            - `fitted` (list): List of fitted model copies (moved to CPU)
191            - `predicted` (pl.DataFrame): Concatenated predictions
192            - `scalers` (list): Feature scalers for each fold
193            - `training_lines` (list): Cell lines used in training
194            - `importance` (None): Not implemented for PyTorch models
195    """
196    assert screen.data is not None, "screen data not loaded"
197    assert screen.metadata is not None, "screen metadata not loaded"
198
199    map_params = screen.params.get("analysis").get("MAP", None)
200    assert map_params is not None, "MAP parameters not found"
201    
202    # Set configs for batch loading 
203    dataloader_config = DataLoaderConfig(**map_params.get("data_loader", {}))
204
205    # Get all cell lines
206    cell_lines = [s["CellLines"].unique() for s in screen.metadata.values()]
207    cell_lines = pl.concat(cell_lines).to_list()
208    cell_lines = list(set(cell_lines) - set(holdout))
209
210    fitted, predicted, scalers_list, split = [], [], [], []
211
212    for cl in cell_lines:
213        # Define test set: current cell line + holdout
214        print(f"---Training hold-out: {cl}---")
215        cell_lines_test = set([cl] + holdout)
216        train_celllines = list(set(cell_lines) - cell_lines_test)
217        split.append(train_celllines)
218        
219        # Train and predict for current holdout cell line
220        fit_model, ypred, scalers = fit_split(
221            screen, model, train_celllines, dataloader_config
222        )
223
224        predicted.append(ypred)
225        fitted_model = getattr(fit_model, 'model', fit_model)
226
227        if isinstance(fitted_model, torch.nn.Module):
228            fitted.append(copy.deepcopy(fitted_model.to("cpu")))
229        else:
230            fitted.append(copy.deepcopy(fitted_model))
231            
232        scalers_list.append(scalers)
233
234    # Use Polars concat for predictions
235    out = {
236        "fitted": fitted,
237        "predicted": pl.concat(predicted),
238        "scalers": scalers_list,
239        "training_lines": cell_lines,
240        "importance": None
241    }
242    
243    return out

281def sample_split(screen: 'ScreenBase', model: BaseModel, holdout: List = [], seed: int = 47) -> Dict:
282    """Wrapper for sample split cross-validation with hold-out cell lines.
283    
284    Splits cell lines 50/50, trains two models reciprocally (each on one split,
285    predicting on the other), and combines predictions. Supports multiple 
286    replicates via params configuration.
287    
288    Args:
289        screen (ScreenBase): Screen object with data and metadata.
290        model (BaseModel): Model instance (SKLearnModel or PyTorchModel).
291        holdout (List[str], optional): Cell lines to exclude from training.
292            Defaults to [].
293        seed (int, optional): Random seed for reproducibility. Defaults to 47.
294        
295    Returns:
296        Dict: Dictionary containing:
297            - `fitted` (list): List of all fitted models from all replicates
298            - `predicted` (pl.DataFrame): Concatenated predictions from replicates
299            - `training_lines` (list): Cell lines in each training split
300            - Additional keys depending on model type
301    """
302    metadata = merge_metadata(screen)
303    holdout += get_mutation_celllines(metadata, ["sporadic"])
304
305    reps = screen.params.get("analysis").get("MAP", {}).get("reps", 1)
306    out_list = []
307    for i in range(reps):
308        print(f"--- Replicate {i+1}/{reps} ---")
309        if isinstance(model, SKLearnModel):
310            out_list.append(sample_split_sklearn(screen, model, holdout, seed + i))
311        elif isinstance(model, PyTorchModel):
312            out_list.append(sample_split_pytorch(screen, model, holdout, seed + i))
313        else:
314            raise ValueError(f"Unsupported model type: {type(model)}")
315
316    # Reformat output list into dict matching sample_split_* structure
317    keys = out_list[0].keys()
318    out = {}
319    for k in keys:
320        if k == 'predicted':
321            dfs = [d[k] for d in out_list if d[k] is not None]
322            out[k] = pl.concat(dfs)
323        elif k == 'importance':
324            out[k] = None
325        else:
326            vals = []
327            for d in out_list:
328                vals.extend(d[k])
329            out[k] = vals
330    
331    return out

334def sample_split_pytorch(
335    screen: "ScreenBase", 
336    model: BaseModel, 
337    holdout: List = [], 
338    seed: int = 47) -> Dict:
339    """Sample split cross-validation for PyTorch models.
340    
341    Performs 50/50 cell line split with reciprocal training using DataLoader
342    batching on single-cell data. Each model is trained on one split and 
343    predicts on the complementary split plus holdout cell lines.
344    
345    Args:
346        screen (ScreenBase): Screen object (ImageScreen or ImageScreenMultiAntibody).
347        model (BaseModel): PyTorchModel instance.
348        holdout (List[str], optional): Cell lines to exclude from training.
349            Defaults to [].
350        seed (int, optional): Random seed for split generation. Defaults to 47.
351        
352    Returns:
353        Dict: Dictionary containing:
354            - `fitted` (list): Two fitted models (moved to CPU)
355            - `predicted` (pl.DataFrame): Concatenated predictions
356            - `training_lines` (list): Cell lines in each training split (2 lists)
357            - `scalers` (list): Feature scalers for each split (2 scalers)
358            - `importance` (None): Not implemented for PyTorch models
359    """
360    assert screen.data is not None, "screen data not loaded"
361    assert screen.metadata is not None, "screen metadata not loaded"
362    assert isinstance(model, PyTorchModel), "model must be a PyTorchModel"
363    
364    map_params = screen.params.get("analysis").get("MAP", None)
365    assert map_params is not None, "MAP parameters not found"
366    dataloader_config = DataLoaderConfig(**map_params.get("data_loader", {}))
367    
368    # Define 50/50 sample split by mutation using cellline_split
369    split = cellline_split(screen, train_prop=0.5, seed=seed, type="CellLines")
370    split1_celllines = split["id_train"].to_series()
371    split2_celllines = split["id_test"].to_series()
372    
373    # Remove holdout cell lines from training sets
374    split1_celllines = list(set(split1_celllines) - set(holdout))
375    split2_celllines = list(set(split2_celllines) - set(holdout))
376
377    # Train on split1, predict on everything except split1
378    fit1, yp1, scalers1 = fit_split(
379        screen, model, split1_celllines, dataloader_config
380    )
381
382    # Train on split2, predict on everything except split2
383    fit2, yp2, scalers2 = fit_split(
384        screen, model, split2_celllines, dataloader_config
385    )
386
387    # Use Polars concat for predictions
388    predicted = pl.concat([yp1, yp2])
389
390    # Safely handle .to('cpu') if available
391    fitted_model1 = getattr(fit1, 'model', fit1)
392    fitted_model2 = getattr(fit2, 'model', fit2)
393
394    if isinstance(fitted_model1, torch.nn.Module):
395        fitted1 = copy.deepcopy(fitted_model1.to("cpu"))
396    else:
397        fitted1 = copy.deepcopy(fitted_model1)
398        
399    if isinstance(fitted_model2, torch.nn.Module):
400        fitted2 = copy.deepcopy(fitted_model2.to("cpu"))
401    else:
402        fitted2 = copy.deepcopy(fitted_model2)
403    
404    out = {
405        'fitted': [fitted1, fitted2],
406        'predicted': predicted,
407        'training_lines': [split1_celllines, split2_celllines],
408        'scalers': [scalers1, scalers2],
409        'importance': None
410    }
411
412    return out
413

415def sample_split_sklearn(
416    screen: 'ScreenBase', 
417    model: BaseModel, 
418    holdout: List = [], 
419    seed: int = 47) -> Dict:
420    """Sample split cross-validation for sklearn models.
421    
422    Performs 50/50 cell line split with reciprocal training on well-averaged
423    features. Each model is trained on one split and predicts on the 
424    complementary split. Predictions are centered around 0.5.
425    
426    Args:
427        screen (ScreenBase): Screen object with data and metadata.
428        model (BaseModel): SKLearnModel instance.
429        holdout (List[str], optional): Cell lines to exclude from training.
430            Defaults to [].
431        seed (int, optional): Random seed for split generation. Defaults to 47.
432        
433    Returns:
434        Dict: Dictionary containing:
435            - `fitted` (list): Two fitted models
436            - `predicted` (pl.DataFrame): Predictions joined with metadata, centered
437            - `importance` (pd.Series or None): Average feature importances
438    """
439    assert screen.data is not None, "screen data not loaded"
440    assert screen.metadata is not None, "screen metadata not loaded"
441    
442    out = {}
443    
444    # Define 50/50 sample split by mutation and set sporadics as hold-out
445    split = cellline_split(screen, train_prop=0.5, seed=seed)
446
447    if holdout is not None:
448        id_holdout = screen.metadata \
449            .filter(pl.col("CellLines").is_in(holdout)) \
450            .select("ID") \
451            .to_series()
452    else:
453        id_holdout = pl.Series([])
454             
455    # Initialize data for model fitting
456    y = screen.get_response() 
457    x = screen.get_data()
458   
459    # Fit models to each sample split
460    id1 = split["id_train"].filter(~split["id_train"].is_in(id_holdout))
461    fit1 = model.fit(x=x, y=y, id_train=id1)
462    
463    id2 = split["id_test"].filter(~split["id_test"].is_in(id_holdout))
464    fit2 = model.fit(x=x, y=y, id_train=id2)
465
466    out['fitted'] = [fit1, fit2]
467    
468    # Generate model predictions
469    id2 = pl.concat([split["id_test"], id_holdout]).unique()
470    yp2 = model.predict(fit1, x, id_test=id2)  
471    yp2 = yp2.with_columns((pl.col("Ypred") - pl.col("Ypred").mean() +  0.5))
472    
473    id1 = pl.concat([split["id_train"], id_holdout]).unique() 
474    yp1 = model.predict(fit2, x, id_test=split['id_train'])
475    yp1 = yp1.with_columns((pl.col("Ypred") - pl.col("Ypred").mean() +  0.5))
476
477    out['predicted'] = pl.concat([yp1, yp2]).join(screen.metadata, on="ID")
478    
479    # Generate model feature importance
480    imp1 = model.get_importance(x)
481    imp2 = model.get_importance(x)
482    if imp1 is not None and imp2 is not None:
483        out['importance'] = (imp1 + imp2) / 2
484    else:
485        out['importance'] = None
486        
487    return out
