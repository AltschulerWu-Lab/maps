maps.models

Model wrappers and implementations used by the MAP pipeline.

This module provides a lightweight abstraction for classification models used to compute MAP scores. Subclasses of BaseModel implement fitting, prediction, and feature-importance extraction for a variety of model types including scikit-learn and PyTorch based implementations.

Module variables:

  • DEFAULT_SEED (int): Default random seed used when not specified in params.

 29class BaseModel():
 30    """Base class for all classification models.
 31
 32    This class serves as a wrapper for fitting models, generating predictions, and
 33    computing feature importances. Subclasses should implement specific model logic
 34    in ``fit()``, ``predict()``, and ``get_importance()`` methods.
 35
 36    Attributes:
 37
 38    - params (dict): Model parameters and configuration.
 39    - model: The underlying model object (implementation-specific).
 40    - fitted (bool): Whether the model has been fitted to data.
 41    """
 42    def __init__(self, **kwargs):
 43        """Initialize a BaseModel instance.
 44
 45        Args:
 46            **kwargs: Arbitrary keyword arguments for model configuration.
 47        """
 48        self.params = kwargs
 49        self.model = None
 50        self.fitted = False
 51    
 52    def fit(self, x=None, y=None, id_train=None, data_loader=None):
 53        """Fit the model to training data.
 54
 55        Args:
 56            x (pl.DataFrame, optional): Feature matrix. Defaults to None.
 57            y (np.ndarray, optional): Response vector. Defaults to None.
 58            id_train (list, optional): IDs of training samples. Defaults to None.
 59            data_loader (DataLoader, optional): PyTorch DataLoader for batch training.
 60                Defaults to None.
 61
 62        Returns:
 63            None
 64
 65        Note:
 66            This method should be implemented by subclasses.
 67        """
 68        raise NotImplementedError()
 69    
 70    def predict(self, x=None, y=None, id_test=None, data_loader=None):
 71        """Generate predictions for test data.
 72
 73        Args:
 74            x (pl.DataFrame, optional): Feature matrix. Defaults to None.
 75            y (np.ndarray, optional): Response vector. Defaults to None.
 76            id_test (list, optional): IDs of test samples. Defaults to None.
 77            data_loader (DataLoader, optional): PyTorch DataLoader for batch prediction.
 78                Defaults to None.
 79
 80        Returns:
 81            pl.DataFrame: Predictions with ID column and probability columns.
 82
 83        Note:
 84            This method should be implemented by subclasses.
 85        """
 86        raise NotImplementedError()
 87    
 88    def get_importance(self, x=None):
 89        """Extract feature importances from the fitted model.
 90
 91        Args:
 92            x (pl.DataFrame, optional): Feature matrix used for training. Defaults to None.

 95            pd.Series or None: Feature importances sorted in descending order, or None

135class Logistic(SKLearnModel):
136    """Logistic regression model for multi-class classification.
137    
138    This model uses scikit-learn's LogisticRegression with feature scaling and
139    balanced sampling during training.
140    
141    Attributes:
142        scaler (StandardScaler): Scaler for feature normalization.
143        model (LogisticRegression): The underlying logistic regression model.
144        fitted (bool): Whether the model has been fitted.
145        scaled (bool): Whether the scaler has been fitted.
146    
147    Example:
148        >>> model = Logistic(model={"Logistic": {"max_iter": 1000}})
149        >>> model.fit(x_train, y_train)
150        >>> predictions = model.predict(x_test)
151    """
152    def __init__(self, **kwargs):
153        """Initialize a Logistic model.
154        
155        Args:
156            **kwargs: Model parameters. Can include a 'model' dict with 'Logistic'
157                or 'BinaryLogistic' keys containing sklearn LogisticRegression parameters.
158        """
159        super().__init__(**kwargs)
160
161        # Initialize model and scaler
162        model_params = self.params.get("model")
163        if model_params is not None and "BinaryLogistic" in model_params:
164            kwargs = model_params["BinaryLogistic"]
165        elif model_params is not None and "Logistic" in model_params:
166            kwargs = model_params["Logistic"]
167        else:
168            kwargs = {}
169        
170        self.scaler = StandardScaler()
171        self.model = LogisticRegression(**kwargs)
172        self.fitted = False
173        self.scaled = False
174    
175    def fit(self, x, y, id_train=None):
176        """Fit the logistic regression model.
177        
178        Args:
179            x (pl.DataFrame): Feature matrix with ID column.
180            y (list): List containing response array(s).
181            id_train (list, optional): IDs of training samples. If None, uses all samples.
182                Defaults to None.
183        """
184        # Filter to training ids if provided
185        if id_train is not None:
186            idx = x["ID"].is_in(id_train)
187            xtrain = x.filter(idx)
188            ytrain = y[0][idx]
189        else:
190            xtrain = x
191            ytrain = y[0]   
192        
193        # sample balance
194        seed = self.params.get("seed", 47)
195        xtrain, ytrain = balanced_sample(xtrain, ytrain, random_state=seed)
196        xtrain = xtrain.drop("ID")
197        
198        # scale feature matrix
199        xtrain = self.scaler.fit_transform(xtrain) # type: ignore
200        self.model.fit(xtrain, ytrain)
201        self.fitted = True
202        self.scaled = True
203        
204
205    def predict(self, x, id_test=None):
206        """Generate probability predictions.
207        
208        Args:
209            x (pl.DataFrame): Feature matrix with ID column.
210            id_test (list, optional): IDs of test samples. If None, uses all samples.
211                Defaults to None.
212        
213        Returns:
214            pl.DataFrame: Predictions with ID and probability columns (Ypred0, Ypred1, etc.).
215        
216        Raises:
217            AssertionError: If model has not been fitted or scaler has not been fitted.
218        """
219        # Check that we received the expected arguments for this model
220        assert self.fitted, "Model must be fitted before prediction"
221        assert self.scaled, "Scaling model must be fitted before prediction"
222
223        # Preprocess the new data
224        if id_test is not None:
225            idx = x["ID"].is_in(id_test)
226            xtest = self.scaler.transform(x.filter(idx).drop("ID"))
227        else:
228            idx = pd.Series([True] * x.shape[0])
229            xtest = self.scaler.transform(x.drop("ID"))
230
231        ypred = self.model.predict_proba(xtest)
232        
233        # Create a dictionary with ID column
234        result_dict = {"ID": x.filter(idx)["ID"]}
235
236        # Add a column for each class probability
237        for i in range(ypred.shape[1]):
238            result_dict[f"Ypred{i}"] = ypred[:, i] 
239
240        return pl.DataFrame(result_dict)
241
242    def get_importance(self, x: pl.DataFrame):
243        # Extract feature importance from the model
244        weights = self.model.coef_[0]
245        features = x.drop("ID").columns
246        importance = pd.Series(weights, index=features)
247        importance = importance.sort_values(ascending=False)
248        
249        return importance

252class BinaryLogistic(Logistic):
253    """Logistic regression for binary classification.
254    
255    Extends the Logistic class but returns only the probability for the positive class.
256    
257    Example:
258        >>> model = BinaryLogistic(model={"BinaryLogistic": {"C": 1.0}})
259        >>> model.fit(x_train, y_train)
260        >>> predictions = model.predict(x_test)  # Returns Ypred column only
261    """
262    def predict(self, x, id_test=None):
263        """Generate binary probability predictions.
264        
265        Args:
266            x (pl.DataFrame): Feature matrix with ID column.
267            id_test (list, optional): IDs of test samples. Defaults to None.
268        
269        Returns:
270            pl.DataFrame: Predictions with ID and Ypred columns (positive class probability).
271        """
272        # Check arguments and delegate to parent
273        result_df = super().predict(x=x, id_test=id_test)
274        result_df = result_df.select(["ID", "Ypred1"])
275        return result_df.rename({"Ypred1": "Ypred"})

280    """Random forest classifier for multi-class classification.
281    
282    This model uses scikit-learn's RandomForestClassifier with balanced sampling

284    
285    Attributes:
286        model (RandomForestClassifier): The underlying random forest model.

289    Example:
290        >>> model = RandomForest(model={"RandomForest": {"n_estimators": 100}})
291        >>> model.fit(x_train, y_train)
292        >>> predictions = model.predict(x_test)
293    """
294    def __init__(self, **kwargs):
295        """Initialize a RandomForest model.
296        
297        Args:
298            **kwargs: Model parameters. Can include a 'model' dict with 'RandomForest'
299                or 'BinaryRandomForest' keys containing sklearn RandomForestClassifier
300                parameters.
301        """
302        super().__init__(**kwargs)
303        
304        # Initialize model
305        model_params = self.params.get("model")
306        if model_params is not None and "RandomForest" in model_params:
307            kwargs = model_params["RandomForest"]
308        elif model_params is not None and "BinaryRandomForest" in model_params:
309            kwargs = model_params["BinaryRandomForest"]
310        else:
311            kwargs = {}
312
313        self.model = RandomForestClassifier(**kwargs)
314        self.fitted = False
315
316    def fit(self, x, y, id_train=None):
317        """Fit the random forest model.
318        
319        Args:
320            x (pl.DataFrame): Feature matrix with ID column.
321            y (list): List containing response array(s).
322            id_train (list, optional): IDs of training samples. If None, uses all samples.
323                Defaults to None.
324        """
325        # Check that we received the expected arguments for this model
326
327        # Filter to training ids if provided
328        if id_train is not None:
329            idx = x["ID"].is_in(id_train)
330            xtrain = x.filter(idx)
331            ytrain = y[0][idx]
332        else:
333            xtrain = x
334            ytrain = y[0]
335
336        # sample balance
337        seed = self.params.get("seed", 47)
338        xtrain, ytrain = balanced_sample(xtrain, ytrain, random_state=seed)
339        xtrain = xtrain.drop("ID")
340
341        # fit model
342        self.model.fit(xtrain, ytrain)
343        self.fitted = True
344
345    def predict(self, x, id_test=None):
346        """Generate class predictions.
347        
348        Args:
349            x (pl.DataFrame): Feature matrix with ID column.
350            id_test (list, optional): IDs of test samples. If None, uses all samples.
351                Defaults to None.
352        
353        Returns:
354            pl.DataFrame: Predictions with ID and Ypred columns (predicted class).
355        
356        Raises:
357            AssertionError: If model has not been fitted.
358        """
359        # Check that we received the expected arguments for this model
360        assert self.fitted, "Model must be fitted before prediction"
361
362        # Preprocess the new data
363        if id_test is not None:
364            idx = x["ID"].is_in(id_test)
365            xtest = x.filter(idx).drop("ID")
366            id_col = x.filter(idx)["ID"]
367        else:
368            xtest = x.drop("ID")
369            id_col = x["ID"]
370
371        ypred = self.model.predict(xtest)
372        return pl.DataFrame({"ID": id_col, "Ypred": ypred})
373    
374    def get_importance(self, x: pl.DataFrame):
375        """Extract feature importances from the random forest.
376        
377        Args:
378            x (pl.DataFrame): Feature matrix with ID column and feature columns.
379        
380        Returns:
381            pd.Series: Feature importances sorted in descending order.
382        
383        Raises:
384            AssertionError: If model has not been fitted.
385        """
386        # Extract feature importance from the model
387        assert self.fitted, "Model must be fitted before getting importance"
388        importance = pd.Series(
389            self.model.feature_importances_,
390            index=x.drop("ID").columns
391        ).sort_values(ascending=False)
392        return importance

420class MultiAntibody(PyTorchModel):
421    """Multi-modal transformer model for multi-antibody imaging data.
422    
423    This model processes data from multiple antibody stains using a transformer-based
424    architecture. It requires a specialized DataLoader that provides batched multi-modal
425    data.
426    
427    Attributes:
428        model (MultiAntibodyClassifier): The underlying PyTorch model.
429        fitted (bool): Whether the model has been fitted.
430    
431    Example:
432        >>> model = MultiAntibody(model={"MultiAntibody": {
433        ...     "model": {"n_classes": 2, "d_model": 16, "n_layers": 1},
434        ...     "train": {"n_epochs": 100}
435        ... }})
436        >>> model.fit(data_loader=train_loader)
437        >>> predictions = model.predict(data_loader=test_loader)
438    """
439    def __init__(self, **kwargs):
440        """Initialize a MultiAntibody model.
441        
442        Args:
443            **kwargs: Model parameters. Should include a 'model' dict with 'MultiAntibody'
444                key containing 'model' (ModelConfig params) and 'train' (TrainConfig params).
445        """
446        super().__init__(**kwargs)
447        self.model = None
448        self.fitted = False
449    
450    def fit(self, data_loader):
451        """Fit the multi-antibody model.
452        
453        Args:
454            data_loader: DataLoader providing multi-modal batched data. Must implement
455                _get_feature_dims() method to determine input dimensions.
456        
457        Raises:
458            ValueError: If data_loader is None.
459        """
460        if data_loader is None:
461            raise ValueError("MultiAntibody requires a data_loader.")
462
463        # Initialize model - requires reading out feature dims from data loader
464        model_params = self.params.get("model")
465        if model_params is not None and "MultiAntibody" in model_params:
466            kwargs = model_params["MultiAntibody"]
467        else:
468            print("No model params found for MultiAntibody, using defaults.")
469            kwargs = {"train": TrainConfig(), "model": ModelConfig()}
470
471        model_config = ModelConfig(**kwargs.get("model", {}))
472        model_config.antibody_feature_dims = data_loader._get_feature_dims() 
473        self.model = MultiAntibodyClassifier(**vars(model_config))
474
475        # Train model 
476        train_config = TrainConfig(**kwargs.get("train", {}))
477        train(self.model, data_loader, train_config)
478        self.fitted = True
479
480    def predict(self, data_loader):
481        """Generate probability predictions for multi-antibody data.
482        
483        Args:
484            data_loader: DataLoader providing multi-modal batched test data.
485        
486        Returns:
487            pl.DataFrame: Predictions with columns for each class probability (Class_0,
488                Class_1, etc.), CellLines, and True labels.
489        
490        Raises:
491            AssertionError: If model has not been fitted or model is None.
492        """
493        assert self.fitted, "Model must be fitted before prediction"
494        assert self.model is not None, "Model must be initialized"
495        
496        self.model.eval()
497        all_probs = []
498        all_labels = []
499        all_lines = []
500        device = next(self.model.parameters()).device
501        data_loader.mode = "eval"
502        
503        with torch.no_grad():
504            for batch in data_loader:
505                if batch is None:
506                    continue
507                x_dict = {ab: batch[ab][0].to(device) for ab in batch}
508                y_line = batch[list(batch.keys())[0]][1].to(device)
509                cl = batch[list(batch.keys())[0]][-1]
510                _, line_logits = self.model(x_dict)
511                probs = torch.softmax(line_logits, dim=1)
512                all_probs.append(probs.cpu())
513                all_labels.append(y_line.cpu())
514
515                if not isinstance(cl, list):
516                    cl = [cl]
517                
518                all_lines.extend(cl)
519            
520        # Merge results into DF
521        all_probs = torch.cat(all_probs, dim=0)
522        all_labels = torch.cat(all_labels, dim=0)
523        preds = pd.DataFrame(all_probs.numpy())
524        preds.columns = [
525            f"Class_{i}" for i in range(self.model.line_head.fc.out_features)
526        ]
527        
528        preds["CellLines"] = all_lines
529        preds["True"] = all_labels.numpy()
530        preds = preds.sort_values(by="Class_0", ascending=False)
531        return pl.DataFrame(preds)
532
533    def _get_importance(self):
534        """Get feature importances.
535        
536        Note:
537            Feature importance extraction is not implemented for deep learning models.
538        
539        Returns:
540            None: Always returns None for this model type.
541        """
542        # Not implemented for deep models
543        return None

153    def __init__(self, **kwargs):
154        """Initialize a Logistic model.
155        
156        Args:
157            **kwargs: Model parameters. Can include a 'model' dict with 'Logistic'
158                or 'BinaryLogistic' keys containing sklearn LogisticRegression parameters.
159        """
160        super().__init__(**kwargs)
161
162        # Initialize model and scaler
163        model_params = self.params.get("model")
164        if model_params is not None and "BinaryLogistic" in model_params:
165            kwargs = model_params["BinaryLogistic"]
166        elif model_params is not None and "Logistic" in model_params:
167            kwargs = model_params["Logistic"]
168        else:
169            kwargs = {}
170

172        self.model = LogisticRegression(**kwargs)
173        self.fitted = False

176    def fit(self, x, y, id_train=None):
177        """Fit the logistic regression model.
178        
179        Args:
180            x (pl.DataFrame): Feature matrix with ID column.
181            y (list): List containing response array(s).
182            id_train (list, optional): IDs of training samples. If None, uses all samples.
183                Defaults to None.
184        """
185        # Filter to training ids if provided
186        if id_train is not None:
187            idx = x["ID"].is_in(id_train)
188            xtrain = x.filter(idx)
189            ytrain = y[0][idx]
190        else:
191            xtrain = x
192            ytrain = y[0]   
193        
194        # sample balance
195        seed = self.params.get("seed", 47)
196        xtrain, ytrain = balanced_sample(xtrain, ytrain, random_state=seed)
197        xtrain = xtrain.drop("ID")
198        
199        # scale feature matrix
200        xtrain = self.scaler.fit_transform(xtrain) # type: ignore
201        self.model.fit(xtrain, ytrain)
202        self.fitted = True
203        self.scaled = True
204        
205
206    def predict(self, x, id_test=None):
207        """Generate probability predictions.
208        
209        Args:
210            x (pl.DataFrame): Feature matrix with ID column.
211            id_test (list, optional): IDs of test samples. If None, uses all samples.
212                Defaults to None.
213        
214        Returns:
215            pl.DataFrame: Predictions with ID and probability columns (Ypred0, Ypred1, etc.).
216        
217        Raises:
218            AssertionError: If model has not been fitted or scaler has not been fitted.
219        """
220        # Check that we received the expected arguments for this model
221        assert self.fitted, "Model must be fitted before prediction"
222        assert self.scaled, "Scaling model must be fitted before prediction"
223
224        # Preprocess the new data
225        if id_test is not None:
226            idx = x["ID"].is_in(id_test)
227            xtest = self.scaler.transform(x.filter(idx).drop("ID"))
228        else:
229            idx = pd.Series([True] * x.shape[0])
230            xtest = self.scaler.transform(x.drop("ID"))
231
232        ypred = self.model.predict_proba(xtest)
233        
234        # Create a dictionary with ID column
235        result_dict = {"ID": x.filter(idx)["ID"]}
236
237        # Add a column for each class probability
238        for i in range(ypred.shape[1]):
239            result_dict[f"Ypred{i}"] = ypred[:, i] 
240
241        return pl.DataFrame(result_dict)
